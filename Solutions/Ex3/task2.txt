WITH MultiHotelSpenders AS (
    SELECT
        c.ID_customer,
        c.name,
        COUNT(b.ID_booking) AS total_bookings,
        COUNT(DISTINCT h.ID_hotel) AS unique_hotels,
        SUM(r.price * (b.check_out_date - b.check_in_date)) AS total_spent
    FROM Customer c
    JOIN Booking b ON c.ID_customer = b.ID_customer
    JOIN Room r ON b.ID_room = r.ID_room
    JOIN Hotel h ON r.ID_hotel = h.ID_hotel
    GROUP BY c.ID_customer
    HAVING COUNT(b.ID_booking) > 2 AND COUNT(DISTINCT h.ID_hotel) > 1
),
BigSpenders AS (
    SELECT
        c.ID_customer,
        c.name,
        SUM(r.price * (b.check_out_date - b.check_in_date)) AS total_spent,
        COUNT(b.ID_booking) AS total_bookings
    FROM Customer c
    JOIN Booking b ON c.ID_customer = b.ID_customer
    JOIN Room r ON b.ID_room = r.ID_room
    GROUP BY c.ID_customer
    HAVING SUM(r.price * (b.check_out_date - b.check_in_date)) > 500
)
SELECT 
    m.ID_customer,
    m.name,
    m.total_bookings,
    m.total_spent,
    m.unique_hotels
FROM MultiHotelSpenders m
JOIN BigSpenders b ON m.ID_customer = b.ID_customer
ORDER BY m.total_spent ASC;
